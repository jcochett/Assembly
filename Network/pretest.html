import socket, sys

# Connect to the server using 'localhost' as the IP, port 6880
s = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
s.connect(('localhost', 6880))

# The server always sends a message size of 64 bytes
# Once connected, the server will ask you for your name
while True:
    data = s.recv(64)
    print data
    if data == "That is correct! Good job!":
        break

    # Enter your name and the server will ask a question specific to you
    message = raw_input("Enter: ")

    # If you enter your name incorrectly, the server will send back an error message
    s.sendall(message)


# Answer the question and the server will tell you if it's correct and close the
# connection or it will ask you to try again if it's incorrect.
s.close()
