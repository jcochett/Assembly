Relevant Topics:    Lab 2: List

the second list and copy the three elements there. If the second list 
in not empty, empty the list before returning the new data

If n is out of range, return ERROR_INVALID_PARAMETER

def getPartialList(listA, n, listB):

    # Validate input
    if type(listA) != list or type(listB) != list:
        return ERROR_INVALID_PARAMETER

    del listB[:]

    if n > (len(listA) - 2) or n < 1:
        return ERROR_INVALID_PARAMETER

    listB.append(listA[n-1])
    listB.append(listA[n])
    listB.append(listA[n+1])

    return ERROR_SUCCESS

Parse the hour of the day from these lines. It will be in hh:mm:ss

def countingTheHours(emailTextList, tupleList):

    # Validate input
    if type(emailTextList) != list or type(tupleList) != list:
        return ERROR_INVALID_PARAMETER

    if len(emailTextList) == 0 or len(tupleList) != 0:
        return ERROR_INVALID_PARAMETER
    
    # Loop input, find From lines, split from line on spaces, split that on :
    elist = []
    for lines in emailTextList:
        if lines.startswith("From ") and lines.find("@"):
            elist.append(re.split(':', re.split(' +', lines)[5])[0])

    # Count list elements and store in a dictionary
    edict = {x:elist.count(x) for x in elist}

    # Store tuple of each dict element into a list
    ehours = [(edict[k], k) for k in edict]

    # Sort tuple and reverse
    ehours.sort()
    ehours.reverse()

    # Add to output tuple
    for x in ehours: tupleList.append(x)
        
    return ERROR_SUCCESS

Read through the list of strings, find each unique letter, convert all

Order the list from the most common letter to the least common letter. 

NOTES:
remove whitespace and puncuation using
string.translate(), string.whitespace, and string.punctuation

Create a list of (letter, count) tuples. Then sort the list in reverse 
order. Assign the sorted tuple to the second parameter.

def countLetters(lines, tupleList):
    
    # Validate input
    if type(lines) != list or type(tupleList) != list:
        return ERROR_INVALID_PARAMETER

    if len(lines) == 0 or len(tupleList) != 0:
        return ERROR_INVALID_PARAMETER
    
    symbol = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ'

    adict = {}

    for key in symbol:
        adict[key] = 0

    # add to output tuple
    for line in lines:                              # Loop though input
        upline = line.upper()                       # Uppercase in-place
        for key in symbol:                          # Loop though my letters
            value = upline.count(key) + adict[key]  # Add previous value to new value stored
            adict.update({key : value})             # Update the dict

    # Store dict keys/values as tuples in a list
    #alist = []
    #for k in adict: alist.append((adict[k], k))
    alist = [(adict[k], k) for k in adict]

    # Sort and Reverse tuple
    alist.sort()
    alist.reverse()

    # Copy list to output list
    for x in alist:
        tupleList.append(x)

    return ERROR_SUCCESS

import argparse


# Recursive fibonacci number
def fib(n): 
    if n==1 or n==2: return 1
    return fib(n-1)+fib(n-2)

def fib_fast(n):
    v1, v2, v3 = 1, 1, 0    # initialise a matrix [[1,1],[1,0]]
    for rec in bin(n)[3:]:  # perform fast exponentiation of the matrix (quickly raise it to the nth power)
        calc = v2*v2
        v1, v2, v3 = v1*v1+calc, (v1+v3)*v2, calc+v3*v3
        if rec=='1':    v1, v2, v3 = v1+v2, v1, v2
    return v2


# Argparse

'''
The sequence of triangle numbers is generated by adding the natural numbers. So the 7th triangle number would be 1 + 2 + 3 + 4 + 5 + 6 + 7 = 28. The first ten terms would be:

1, 3, 6, 10, 15, 21, 28, 36, 45, 55, ...

Let us list the factors of the first seven triangle numbers:

     1: 1
     3: 1,3
     6: 1,2,3,6
    10: 1,2,5,10
    15: 1,3,5,15
    21: 1,3,7,21
    28: 1,2,4,7,14,28

We can see that 28 is the first triangle number to have over five divisors.

What is the value of the first triangle number to have over five hundred divisors?
lower range x by 2
'''
import time
from math import sqrt 
  
# Calculate triangle
def triangle(n):
    i = 0
    while i < n:
        yield i
        i += 1
         



def factors1(n):    
    return set(reduce(list.__add__, ([i, n//i] for i in range(1, int(n**0.5) + 1) if n % i == 0)))


def factors2(n):
    return set(x for tup in ([i, n//i] 
        for i in range(1, int(n**0.5)+1) if n % i == 0) for x in tup)

def factors3(n):
        step = 2 if n%2 else 1
        return set(reduce(list.__add__, 
                    ([i, n//i] for i in range(1, int(sqrt(n))+1, step) if n % i == 0)))

start = time.time()

    x = len(factors3(n))

        print "Number: ", n, " with ", x, " divisors at round: ", i
        break

import threading, thread

g_counter = 0
g_lock = threading.Lock()

class ThreadClass(object):
    ''' docstring for ThreadClass class '''
    def __init__(self, tnum):
        self.tnum = tnum

    def inc_counter(self, times=1000, lock=False):
        global g_counter
        for i in xrange(self.tnum):
            if lock:
                g_lock.acquire()
                g_counter += 1
                print "\r", g_counter,
                g_lock.release()
            else:
                g_counter += 1

    def dec_counter(self, times=1000, lock=False):
        global g_counter
        for i in xrange(self.tnum):
            if lock:
                g_lock.acquire()
                g_counter -= 1
                print "\r", g_counter,
                g_lock.release()
            else:
                g_counter -= 1

    def run_threads(self, num_threads = 1000, *args, **kwargs):
        global g_counter
        global_counter = 0
        print "Global Counter start: ", g_counter
        threads = [threading.Thread(target=self.inc_counter, args=args, kwargs=kwargs) for i in xrange(self.tnum)]
        for thread in threads:
            thread.start()
        for thread in threads:
            thread.join()
        print "Global Counter end: ", g_counter
        threads = [threading.Thread(target=self.dec_counter, args=args, kwargs=kwargs) for i in xrange(self.tnum)]
        for thread in threads:
            thread.start()
        for thread in threads:
            thread.join()
